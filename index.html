Exemplu de cod

app.zing 
import "json";
import "crypto";

let request_json = "{\"user\": \"Mihai\", \"pass\": \"secret123\"}";
print("=== Request JSON primit ===");

let data = json.parse(request_json);
print("Utilizator:", data.user);
print("Parola (plaintext):", data.pass);

let hash = crypto.hash_password(data.pass);
print("Hash generat:", hash);

let corecta = crypto.verify_password(data.pass, hash);
print("Parola potrivita?", corecta);
